#include <stdio.h>

int main() { (INÍCIO)
 
    int num1, num2, soma; (VARIAVEIS)
    
    printf("Informe Primeiro Número: "); (ESCREVA)
    scanf("%d", &num1); (LEIA)

    soma = num1 + num2; (DESCREVE COMO VAI SER FEITO)
    
    printf("A soma é: %d", soma); (ESCREVA)
    
    return 0; (FIM)
}
